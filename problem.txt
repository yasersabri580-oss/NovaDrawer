class _DashboardState extends State<Dashboard> with TickerProviderStateMixin {
  late NovaDrawerController _drawerController;
  String _selectedPageTitle = '';
  final NovaHeaderVariant _headerVariant = NovaHeaderVariant.classic;
  int _tabIndex = 0;
  late TabController _tabController;
  bool _initialized = false;
  @override
  void initState() {
    super.initState();
    //if was desktop opened with a tab index, keep it when coming back from another page

    _tabController = TabController(length: 5, vsync: this);
    init();
  }

  Future<void> init() async {}

  @override
  void didChangeDependencies() {
    super.didChangeDependencies();
    final l10n = AppLocalizations.of(context);
    if (_selectedPageTitle.isEmpty) {
      _selectedPageTitle = l10n?.home ?? 'Home';
    }

    if (_initialized) return;

    final isDesktop = ResponsiveHelper.of(context).isDesktop;

    if (isDesktop) {
      _tabIndex = 0;
      _drawerController = NovaDrawerController(initiallyOpen: true);
    } else {
      _drawerController = NovaDrawerController(initiallyOpen: false);
    }
  }

  @override
  void dispose() {
    _drawerController.dispose();
    _tabController.dispose();
    _backTimer?.cancel();
    super.dispose();
  }

  bool canPop = false;

  Timer? _backTimer;

  Future<void> _handleBackPressed() async {
    // first press
    if (!canPop) {
      setState(() {
        canPop = true;
      });

      _backTimer?.cancel();

      _backTimer = Timer(const Duration(seconds: 2), () {
        if (mounted) {
          setState(() {
            canPop = false;
          });
        }
      });

      return;
    }
    ToastKit.channel(AppToastChannel.app.name).show(
      ToastEvent.info(
        message: "آیا میخواهید از برنامه خارج شوید؟",
        dismissible: false,
        actions: [
          ToastAction(
            label: "بله",
            onPressedWithController: (controller) {
              controller.dismiss();
              SystemNavigator.pop();
            },
          ),
          ToastAction(
            label: "خیر",
            onPressedWithController: (controller) {
              controller.dismiss();
              setState(() {
                canPop = false;
              });
            },
          ),
        ],
        customVariantName: AppToastVariant.appInfo.name,
      ),
    );
  }

  void _onItemTap(NovaDrawerItem item) {
    setState(() {
      _selectedPageTitle = item.title;
    });
    switch (item.id) {
      case 'logout':
        context.read<AuthBloc>().add(LogoutEvent());
        GoRouter.of(context).go(AppRouter.loginPage);
        break;
      case 'about':
        GoRouter.of(context).go(AppRouter.aboutPage);
        break;
      default:
        if (item.route != null && item.route!.isNotEmpty) {
          GoRouter.of(context).push(item.route!);
        }
    }
  }

  @override
  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context);
    final colorScheme = Theme.of(context).colorScheme;
    final drawerTheme =
        context.read<ThemeCubit>().state.variant == ThemeVariant.dark
        ? NovaDrawerTheme.dark()
        : NovaDrawerTheme.light();

    final config = NovaDrawerConfig(
      enableAutoScrollToSelected: true,
      autoScrollDuration: const Duration(milliseconds: 200),
      autoScrollCurve: Curves.easeInOut,
      animationType: NovaDrawerAnimationType.slide,
      displayMode: NovaDrawerDisplayMode.auto,
      showMiniOnCollapse: true,
      gestureConfig: const NovaDrawerGestureConfig(
        enableSwipeToOpen: true,
        enableSwipeToClose: true,
      ),
      animationConfig: const NovaDrawerAnimationConfig(
        duration: Duration(milliseconds: 350),
        curve: Curves.easeOutCubic,
        enableStaggeredAnimations: true,
      ),
    );

    final headerConfig = NovaHeaderConfig(
      variant: _headerVariant,
      profile: NovaHeaderUserProfile(
        name: 'accounting_helal_front User',
        email: 'user@accounting_helal_front.me',
        status: NovaUserStatus.online,
        notificationCount: 0,
      ),
      showCloseButton: true,
      showPinButton: false,
      showStatusIndicator: true,
      enableCollapseExpand: true,
      gradientColors: const [
        Color(0xFF1565C0),
        Color(0xFF0288D1),
        Color(0xFF00ACC1),
      ],
      actions: [
        NovaHeaderAction(
          id: 'notifications',
          icon: Icons.notifications_outlined,
          tooltip: l10n?.notifications ?? 'Notifications',
          onTap: () {},
        ),
        if (ResponsiveHelper.of(context).isDesktop)
          NovaHeaderAction(
            customWidget: BlocBuilder<ThemeCubit, ThemeState>(
              builder: (context, themeState) {
                final isDark = themeState.variant == ThemeVariant.dark;
                return IconButton(
                  icon: Icon(isDark ? Icons.light_mode : Icons.dark_mode),
                  onPressed: () => context.read<ThemeCubit>().toggle(),
                  tooltip: isDark ? 'Light Mode' : 'Dark Mode',
                );
              },
            ),
            id: 'theme',
            icon: Icons.abc,
          ),
      ],
      onEditProfile: () {},
      onSwitchAccount: () {},
    );

    // Tab labels built from l10n so they are always localised.
    final tabLabels = [
      l10n?.shopping ?? 'Shopping',
      l10n?.debt ?? 'Debt',
      l10n?.balanceSheet ?? 'Balance Sheet',
      l10n?.inventory ?? 'Inventory',
      l10n?.users ?? 'Users',
    ];

    return BlocBuilder<UserPermissionBloc, UserPermissionState>(
      builder: (context, permState) {
        UserPermissionModel? userPerm;
        if (permState is UserPermissionListLoadedState) {
          final authState = context.read<AuthBloc>().state;
          int? currentUserId;
          if (authState is AuthLoginSuccessState) {
            currentUserId = authState.data.userId;
          }
          try {
            userPerm = permState.items.firstWhere(
              (p) => p.userId == currentUserId,
            );
          } on StateError {
            userPerm = null;
          }
        } else {
          userPerm = null;
        }

        final sections = _buildSections(l10n, userPerm);

        final drawer = NovaAppDrawer(
          onNavigate: novaDrawerBodyNavigate(
            pages: pages,
            external: (context, route) => GoRouter.of(context).push(route),
          ),
          controller: _drawerController,
          entries: [
            NovaDrawerItemEntry(
              NovaDrawerItem(
                id: 'home',
                title: l10n?.home ?? 'Home',
                icon: Icons.home_outlined,
                selectedIcon: Icons.home,
                route: '/home',
              ),
            ),

            ...sections.map((s) => NovaDrawerSectionEntry(s)),

            NovaDrawerItemEntry(
              NovaDrawerItem(
                id: 'logout',
                title: l10n?.logout ?? 'Logout',
                icon: Icons.logout_outlined,
                selectedIcon: Icons.logout,
                route: '/logout',
              ),
            ),

            NovaDrawerItemEntry(
              NovaDrawerItem(
                id: 'about',
                title:
                    l10n?.aboutaccounting_helal_front ??
                    'About accounting_helal_front',
                icon: Icons.info_outline,
                selectedIcon: Icons.info,
                route: '/about',
              ),
            ),
          ],
          sections: sections,
          header: NovaDrawerHeader(config: headerConfig),
          footer: _buildDrawerFooter(l10n),
          onItemTap: _onItemTap,
          theme: drawerTheme,
          config: config,
        );

        return PopScope(
          canPop: canPop,
          onPopInvokedWithResult: (didPop, result) async {
            if (didPop) return;
            await _handleBackPressed();
          },
          child: NovaDrawerScaffold(
            bottomNavigationBar: !ResponsiveHelper.of(context).isDesktop
                ? NovaBottomNavigationBar(
                    labels: tabLabels,
                    icons: const [
                      Icons.shopping_cart_outlined,
                      Icons.account_balance_wallet_outlined,
                      Icons.grid_view_outlined,
                      Icons.inventory_2_outlined,
                      Icons.people_outlined,
                    ],
                    initialSelectedTab: tabLabels[_tabIndex],
                    tabSize: 50,
                    tabBarHeight: 65,
                    textStyle: TextStyle(
                      fontSize: 11,
                      color: colorScheme.onPrimary,
                      fontWeight: FontWeight.w500,
                    ),
                    tabIconColor: colorScheme.onSurfaceVariant,
                    tabSelectedColor: colorScheme.primary,
                    onTabItemSelected: (int value) {
                      _drawerController.selectItem('');
                      setState(() {
                        _tabIndex = value;
                      });
                    },
                  )
                : null,

            controller: _drawerController,
            drawer: drawer,
            theme: drawerTheme,
            config: config,
            onItemTap: _onItemTap,
            miniDrawerHeader: Icon(
              Icons.account_balance_outlined,
              color: colorScheme.primary,
            ),
            appBar: !ResponsiveHelper.of(context).isDesktop
                ? AppBar(
                    actions: [
                      BlocBuilder<ThemeCubit, ThemeState>(
                        builder: (context, themeState) {
                          final isDark =
                              themeState.variant == ThemeVariant.dark;
                          return IconButton(
                            icon: Icon(
                              isDark ? Icons.light_mode : Icons.dark_mode,
                            ),
                            onPressed: () =>
                                context.read<ThemeCubit>().toggle(),
                            tooltip: isDark ? 'Light Mode' : 'Dark Mode',
                          );
                        },
                      ),
                    ],
                  )
                : null,
            body: NovaDrawerBodyRouter(
              pages: pages,
              controller: _drawerController,
              fallback: ResponsiveHelper.of(context).isDesktop
                  ? null
                  : IndexedStack(
                      index: _tabIndex,
                      children: [
                        _TabPlaceholder(
                          icon: Icons.shopping_cart_outlined,
                          label: tabLabels[0],
                        ),
                        _TabPlaceholder(
                          icon: Icons.account_balance_wallet_outlined,
                          label: tabLabels[1],
                        ),
                        _TabPlaceholder(
                          icon: Icons.grid_view_outlined,
                          label: tabLabels[2],
                        ),
                        _TabPlaceholder(
                          icon: Icons.inventory_2_outlined,
                          label: tabLabels[3],
                        ),
                        _TabPlaceholder(
                          icon: Icons.people_outlined,
                          label: tabLabels[4],
                        ),
                      ],
                    ),
            ),
          ),
        );
      },
    );
  }

  Widget _buildDrawerFooter(AppLocalizations? l10n) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      mainAxisSize: MainAxisSize.min,
      children: [
        const Divider(height: 1),
        Padding(
          padding: const EdgeInsets.all(8.0),
          child: Text(
            '${l10n?.appVersion ?? 'App Version'} 1.0.0',
            style: TextStyle(
              fontSize: 12,
              color: Theme.of(context).colorScheme.onSurfaceVariant,
            ),
          ),
        ),
      ],
    );
  }
}

/// Placeholder widget shown inside each tab until real pages are implemented.
class _TabPlaceholder extends StatelessWidget {
  const _TabPlaceholder({required this.icon, required this.label});

  final IconData icon;
  final String label;

  @override
  Widget build(BuildContext context) {
    return Center(
      child: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(icon, size: 64, color: Theme.of(context).colorScheme.primary),
          const SizedBox(height: 16),
          Text(label, style: Theme.of(context).textTheme.headlineSmall),
        ],
      ),
    );
  }
}
